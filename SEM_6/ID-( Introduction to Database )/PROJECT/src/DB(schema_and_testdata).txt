CREATE TABLE bank_customers (
    custno VARCHAR2(10) PRIMARY KEY,
    name VARCHAR2(50) NOT NULL,
    phoneno VARCHAR2(15) NOT NULL,
    city VARCHAR2(30) NOT NULL
);

CREATE TABLE bank_branch (
    branchcode VARCHAR2(10) PRIMARY KEY,
    branchname VARCHAR2(50) NOT NULL,
    branchcity VARCHAR2(30)
);

CREATE TABLE bank_account (
    accno VARCHAR2(10) PRIMARY KEY,
    custno VARCHAR2(10) REFERENCES bank_customers(custno),
    acctype VARCHAR2(20),
    balance NUMBER(15,2) NOT NULL,
    branchcode VARCHAR2(10) REFERENCES bank_branch(branchcode)
);

CREATE TABLE bank_loan (
    loanno VARCHAR2(10) PRIMARY KEY,
    custno VARCHAR2(10) REFERENCES bank_customers(custno),
    loanamount NUMBER(15,2),
    branchcode VARCHAR2(10) REFERENCES bank_branch(branchcode),
    loantype VARCHAR2(30),
    interestrate NUMBER(5,2),
    startdate DATE,
    enddate DATE
);
